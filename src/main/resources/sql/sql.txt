#创建库
CREATE DATABASE REIZ;
#使用库
USE REIZ;
#用户表
CREATE TABLE TB_USER(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	USER_KIND INT COMMENT '用户类型#1:商家;2:买家;',
	USER_NAME VARCHAR(20) NOT NULL  COMMENT '用户名称',
	USER_CODE VARCHAR(20) NOT NULL UNIQUE COMMENT '用户账号',
	USER_PASS VARCHAR(60) NOT NULL COMMENT '用户账号',
	USER_PHONE VARCHAR(15) COMMENT '用户手机',
	IS_LOCK INT COMMENT '是否锁定#1:是;0:否;',
	LAST_LOGIN_IP VARCHAR(20) COMMENT '最后的登录IP',
	LAST_LOGIN_DATE DATETIME COMMENT '最后的登录时间',
	ACTIVE_FLAG INT COMMENT'数据是否有效#1:是;0:否;',
	CREATE_BY VARCHAR (10)COMMENT'创建人',
	CREATE_DATE DATETIME COMMENT'创建时间',
	UPDATE_BY VARCHAR(10) COMMENT'修改人',
	UPDATE_DATE DATETIME COMMENT'修改时间'	
);

#目录表
CREATE TABLE TB_MENU(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	PARENT_ID BIGINT COMMENT '父类主键',
	MENU_KIND INT COMMENT '目录级别',
	MENU_NAME VARCHAR(30) NOT NULL  COMMENT '目录名称',
	MENU_INFO VARCHAR(60) NOT NULL COMMENT '目录信息',
	ACTIVE_FLAG INT COMMENT'数据是否有效#1:是;0:否;',
	CREATE_BY VARCHAR (10)COMMENT'创建人',
	CREATE_DATE DATETIME COMMENT'创建时间',
	UPDATE_BY VARCHAR(10) COMMENT'修改人',
	UPDATE_DATE DATETIME COMMENT'修改时间'	
);

#收货地址名称表
CREATE TABLE TB_ADDRESS(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	USER_CODE VARCHAR(30) COMMENT '用户CODE',
	AREA1 INT COMMENT '省CODE',
	AREA2 INT COMMENT '市CODE',
	AREA3 INT COMMENT '区CODE',
	LINK_NAME VARCHAR(30) COMMENT '收货人姓名',
	LINK_PHONE VARCHAR(30) COMMENT'收货人电话',
	LINK_ADDR VARCHAR(120) COMMENT'详细地址',
	POST_CODE VARCHAR(20) COMMENT'邮编',
	IS_DEFAULT INT COMMENT '是否默认#1:是;0:否;',
	ACTIVE_FLAG INT COMMENT'数据是否有效#1:是;0:否;',
	CREATE_BY VARCHAR (10)COMMENT'创建人',
	CREATE_DATE DATETIME COMMENT'创建时间',
	UPDATE_BY VARCHAR(10) COMMENT'修改人',
	UPDATE_DATE DATETIME COMMENT'修改时间'	
);
#商品表
CREATE TABLE TB_PRODUCT(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	MENU_ID BIGINT COMMENT '目录ID',
	PRO_CODE VARCHAR(30) COMMENT '商品编号',
	PRO_NAME VARCHAR(30) COMMENT'商品名称',
	PRO_PATH VARCHAR(120) COMMENT'商品路径',
	PRO_PRICE FLOAT  COMMENT'商品价格',
	IS_LINE INT COMMENT '是否上架#1:是;0:否;',
	PRO_INFO TEXT COMMENT '商品简介',
	ACTIVE_FLAG INT COMMENT'数据是否有效#1:是;0:否;',
	CREATE_BY VARCHAR (10)COMMENT'创建人',
	CREATE_DATE DATETIME COMMENT'创建时间',
	UPDATE_BY VARCHAR(10) COMMENT'修改人',
	UPDATE_DATE DATETIME COMMENT'修改时间'	
);
