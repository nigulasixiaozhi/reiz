CREATE OR REPLACE VIEW V_PRODUCT AS
SELECT
  P.ROW_ID,
  MENU_ID,
  M.MENU_NAME,
  PRO_CODE,
  PRO_NAME,
  PRO_PATH,
  OLD_PRICE,
  NEW_PRICE,
  PRO_COUNT,
  IS_NEW,
  IS_LINE,
  PRO_INFO,
  P.ACTIVE_FLAG,
  P.CREATE_BY,
  P.CREATE_DATE,
  P.UPDATE_BY,
  P.UPDATE_DATE
FROM
  TB_PRODUCT P
  LEFT JOIN TB_MENU M
    ON P.MENU_ID = M.ROW_ID
